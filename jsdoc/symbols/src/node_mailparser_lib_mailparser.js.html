 19 // MailParser is a FSM - it is always in one of the possible states
 20 var STATES = {
 21     header:   0x1,
 22     body:     0x2,
 23     finished: 0x3
 24 };

 26 /**
 27  * Creates instance of MailParser which in turn extends Stream
 28  * 
 29  * @constructor
 30  * @param {Object} [options] Optional options object
 31  */
 32 function MailParser(options){
 33     
 34     // Make MailParser a Stream object
 35     Stream.call(this);
 36     this.writable = true;
 37     
 38     /** @private*/ this.options = options || {};
 39      
 40     /** @private */ this.state         = STATES.header;
 41     /** @private */ this.remainder     = "";
 42     /** @public  */ this.mimeTree      = this.createMimeNode();
 43     /** @private */ this.currentNode   = this.mimeTree;
 44     /** @private */ this.fileNames     = {};
 45     /** @private */ this.multipartTree = [];
 46     /** @private */ this.iconv         = {};
 47     /** @private */ this.mailData      = {};
 48 }
 49 // inherit methods and properties of Stream
 50 utillib.inherits(MailParser, Stream);

 52 /**
 53  * Writes a value to the MailParser stream
 54  * 
 55  * @param {Buffer|String} chunk The data to be written to the MailParser stream
 56  * @param {String} [encoding] The encoding to be used when "chunk" is a string
 57  * @returns {Boolean} Returns true if nothing is buffered, for later draining returns false
 58  */
 59 MailParser.prototype.write = function(chunk, encoding){
 60     if(typeof chunk == "string"){
 61         chunk = new Buffer(chunk, encoding);
 62     }
 63     
 64     if(chunk && chunk.length){
 65         this.remainder += chunk.toString("binary");
 66         // to slow things down cache the input and process a bit later,
 67         // emit "drain" when data is processed
 68         process.nextTick(this.process.bind(this));
 69         return false;
 70     }else{
 71         return true;
 72     }
 73 };

 75 /**
 76  * Terminates the MailParser stream 
 77  * 
 78  * If "chunk" is set, writes it to the Stream before terminating.
 79  * 
 80  * @param {Buffer|String} chunk The data to be written to the MailParser stream
 81  * @param {String} [encoding] The encoding to be used when "chunk" is a string
 82  */
 83 MailParser.prototype.end = function(chunk, encoding){
 84     if(typeof chunk == "string"){
 85         chunk = new Buffer(chunk, encoding);
 86     }
 87     if(chunk){
 88         this.remainder += chunk.toString("binary");
 89     }
 90 
 91     process.nextTick(this.process.bind(this, true));
 92 };

 94 /**
 95  * Processes the data written to the MailParser stream
 96  * 
 97  * The data is split into lines and each line is processed individually. Last
 98  * line in the batch is preserved as a remainder since it is probably not a
 99  * complete line but just the beginning of it. The remainder is later prepended
100  * to the next batch of data.
101  * 
102  * If "lastLine" is set to true, 
103  * 
104  * @param {Boolean} [finalPart=false] if set to true indicates that this is the last part of the stream
105  */
106 MailParser.prototype.process = function(finalPart){
107     
108     finalPart = !!finalPart;
109     
110     var lines = this.remainder.split(/\r?\n|\r/),
111         line, i, len;
112         
113     if(!finalPart){
114         this.remainder = lines.pop();
115         // force line to 1MB chunks if needed 
116         if(this.remainder.length>1048576){
117             this.remainder = this.remainder.replace(/(.{1048576}(?!\r?\n|\r))/g,"$&\n");
118         }
119     }
120     
121     for(i=0, len=lines.length; i < len; i++){
122         line = lines[i];
123         
124         if(this.state == STATES.header){
125             if(this.processStateHeader(line) === true){
126                 continue;
127             }
128         }
129         
130         if(this.state == STATES.body){
131             
132             if(this.processStateBody(line) === true){
133                 continue;
134             }
135             
136         }
137     }
138     
139     if(!finalPart){
140         process.nextTick(this.emit.bind(this, "drain"));
141     }else{
142         this.state = STATES.finished;
143         process.nextTick(this.processMimeTree.bind(this));
144     }
145 };

147 /**
148  * Processes a line while in header state
149  * 
150  * If header state ends and body starts, detect if the contents is an attachment
151  * and create a stream for it if needed
152  * 
153  * @param {String} line The contents of a line to be processed
154  * @returns {Boolean} If state changes to body retuns true
155  */
156 MailParser.prototype.processStateHeader = function(line){
157     var boundary, i, len, attachment, 
158         lastPos = this.currentNode.headers.length - 1;
159     
160     // Check if the header endas and body starts
161     if(!line.length){
162         this.state = STATES.body;
163         
164         // if there's unprocessed header data, do it now
165         if(lastPos >= 0){
166             this.processHeaderLine(lastPos);
167         }
168         
169         // handle attachment start
170         if((["text/plain", "text/html"].indexOf(this.currentNode.meta.contentType || "") < 0 || 
171           ["attachment", "inline"].indexOf(this.currentNode.meta.contentDisposition)>=0) && 
172           !this.currentNode.meta.mimeMultipart){
173             
174             this.currentNode.attachment = true;
175             
176             fileName = this.currentNode.meta.fileName || "attachment";
177             this.currentNode.meta.generatedFileName = this.generateFileName(fileName);
178             
179             attachment = {
180                 fileName: fileName
181             };
182             
183             if(this.options.streamAttachments){
184                 if(this.currentNode.meta.contentType){
185                     attachment.contentType = this.currentNode.meta.contentType || null;
186                 }
187                 
188                 if(this.currentNode.meta.contentId){
189                     attachment.contentId = this.currentNode.meta.contentId || "";
190                 }
191                 
192                 if(this.currentNode.meta.charset){
193                     attachment.charset = this.currentNode.meta.charset || "utf-8";
194                 }
195             
196                 if(this.currentNode.meta.transferEncoding == "base64"){
197                     this.currentNode.stream = new Streams.Base64Stream();
198                 }else if(this.currentNode.meta.transferEncoding == "quoted-printable"){
199                     this.currentNode.stream = new Streams.QPStream(this.currentNode.meta.charset);
200                 }else{
201                     this.currentNode.stream = new Streams.BinaryStream();
202                 }
203                 attachment.stream = this.currentNode.stream;
204                 
205                 this.emit("attachment", attachment);
206             }else{
207                 this.currentNode.content = "";
208             }
209         }
210         
211         return true;
212     }
213     
214     // unfold header lines if needed
215     if(i && line.match(/^\s+/) && lastPos>=0){
216         this.currentNode.headers[lastPos] += " " + line.trim();
217     }else{
218         this.currentNode.headers.push(line.trim());
219         if(lastPos>=0){
220             // if a complete header line is received, process it
221             this.processHeaderLine(lastPos);
222         }
223     }
224     
225     return false;
226 };

228 /**
229  * Processes a line while in body state
230  * 
231  * @param {String} line The contents of a line to be processed
232  * @returns {Boolean} If body ends return true
233  */
234 MailParser.prototype.processStateBody = function(line){
235     var i, len, node,
236         nodeReady = false;
237     
238     // Handle multipart boundaries
239     if(line.substr(0, 2) == "--"){
240         for(i=0, len = this.multipartTree.length; i<len; i++){
241             
242             // check if a new element block starts
243             if(line == "--" + this.multipartTree[i].boundary){
244                 
245                 if(this.currentNode.content || this.currentNode.stream){
246                     this.finalizeContents();
247                 }
248                 
249                 node = this.createMimeNode(this.multipartTree[i].node);
250                 this.multipartTree[i].node.childNodes.push(node);
251                 this.currentNode = node;
252                 this.state = STATES.header;
253                 nodeReady = true;
254                 break;
255             }else 
256             // check if a multipart block ends
257               if(line == "--" + this.multipartTree[i].boundary + "--"){
258                 
259                 if(this.currentNode.content || this.currentNode.stream){
260                     this.finalizeContents();
261                 }
262                 
263                 if(this.multipartTree[i].node.parentNode){
264                     this.currentNode = this.multipartTree[i].node.parentNode;
265                 }else{
266                     this.currentNode = this.multipartTree[i].node;
267                 }
268                 this.state = STATES.body;
269                 nodeReady = true;
270                 break;
271             }
272         }
273     }
274     if(nodeReady){
275         return true;
276     }
277     
278     // handle text or attachment line
279     if(["text/plain", "text/html"].indexOf(this.currentNode.meta.contentType || "")>=0 && 
280       !this.currentNode.attachment){
281         this.handleTextLine(line);
282     }else if(this.currentNode.attachment){
283         this.handleAttachmentLine(line);
284     }
285     
286     return false;
287 };

289 /**
290  * Processes a complete unfolded header line
291  * 
292  * Processes a line from current node headers array and replaces its value.
293  * Input string is in the form of "X-Mailer: PHP" and its replacement would be
294  * an object {key: "x-mailer", value: "PHP"}
295  * 
296  * Additionally node meta object will be filled also, for example with data from
297  * To: From: Cc: etc fields.
298  * 
299  * @param {Number} pos Which header element (from an header lines array) should be processed
300  */
301 MailParser.prototype.processHeaderLine = function(pos){
302     var key, value, parts, line;
303     
304     pos = pos || 0;
305     
306     if(!(line = this.currentNode.headers[pos])){
307         return;
308     }
309     
310     parts = line.split(":");
311     
312     key = parts.shift().toLowerCase().trim();
313     value = parts.join(":").trim();
314     
315     switch(key){
316         case "content-type":
317             value = this.parseContentType(value);
318             break;
319         case "mime-version":
320             this.currentNode.useMIME = true;
321             break;
322         case "date":
323             this.currentNode.meta.date = new Date(datetime.strtotime(value)*1000 || Date.now());
324             break;
325         case "to":
326             this.currentNode.to = mimelib.parseAddresses(value);
327             break;
328         case "from":
329             this.currentNode.from = mimelib.parseAddresses(value);
330             break;
331         case "cc":
332             this.currentNode.cc = mimelib.parseAddresses(value);
333             break;
334         case "bcc":
335             this.currentNode.bcc = mimelib.parseAddresses(value);
336             break;
337         case "x-priority":
338         case "x-msmail-priority":
339         case "importance":
340             value = this.parsePriority(value);
341             break;
342         case "message-id":
343             this.currentNode.meta.messageId = this.trimQuotes(value);
344             break;
345         case "references":
346             this.currentNode.meta.messageReferences = this.trimQuotes(value);
347             break;
348         case "in-reply-to":
349             this.currentNode.meta.inReplyTo = this.trimQuotes(value);
350             break;
351         case "thread-index":
352             this.currentNode.meta.threadIndex = value;
353             break;
354         case "content-transfer-encoding":
355             this.currentNode.meta.transferEncoding = value.toLowerCase();
356             break;
357         case "subject":
358             this.currentNode.subject = this.encodeString(value);
359             break;
360         case "content-disposition":
361             this.parseContentDisposition(value);
362             break;
363         case "content-id":
364             this.currentNode.meta.contentId = this.trimQuotes(value);
365             break;
366     }
367     
368     this.currentNode.headers[pos] = {key: key, value: value};
369 };

371 /**
372  * Creates an empty node element for the mime tree
373  * 
374  * Created element includes parentNode property and a childNodes array. This is
375  * needed to later walk the whole mime tree
376  * 
377  * @param {Object} [parentNode] the parent object for the created node
378  * @returns {Object} node element for the mime tree
379  */
380 MailParser.prototype.createMimeNode = function(parentNode){
381     var node = {
382         parentNode: parentNode || this.currentNode || null,
383         headers: [],
384         meta: {},
385         childNodes: []
386     };
387     
388     return node;
389 };

391 /**
392  * Splits a header value into key-value pairs
393  * 
394  * Splits on ";", the first value will be set as defaultValue property and will
395  * not be handled, others will be split on "=" to key-value pairs
396  * 
397  * For example
398  * 
399  *     content-type: text/plain; charset=utf-8
400  * 
401  * Will become
402  * 
403  *     {
404  *         defaultValue: "text/plain",
405  *         charset: "utf-8"
406  *     }
407  * 
408  * @param {String} value A string to be splitted into key-value pairs
409  * @returns {Object} a key-value object, with defaultvalue property
410  */
411 MailParser.prototype.parseHeaderLineWithParams = function(value){
412     var key, parts, returnValue = {};
413     
414     parts = value.split(";");
415     returnValue.defaultValue = parts.shift().toLowerCase();
416     
417     for(var i=0, len = parts.length; i<len; i++){
418         value = parts[i].split("=");
419         key = value.shift().trim();
420         value = value.join("=").trim();
421         
422         // trim quotes
423         value = this.trimQuotes(value);
424         returnValue[key] = value;
425     }
426     
427     return returnValue;
428 };

430 /**
431  * Parses a Content-Type header field value
432  * 
433  * Fetches additional properties from the content type (charset etc.) and fills
434  * current node meta object with this data
435  * 
436  * @param {String} value Content-Type string
437  * @returns {Object} parsed contenttype object
438  */
439 MailParser.prototype.parseContentType = function(value){
440     value = this.parseHeaderLineWithParams(value);
441     
442     if(value){
443         if(value.defaultValue){
444             value.defaultValue = value.defaultValue.toLowerCase();
445             this.currentNode.meta.contentType = value.defaultValue;
446             if(value.defaultValue.substr(0,"multipart/".length)=="multipart/"){
447                 this.currentNode.meta.mimeMultipart = value.defaultValue.substr("multipart/".length);
448             }
449         }else{
450             this.currentNode.meta.contentType = "application/octet-stream";
451         }
452         if(value.charset){
453             value.charset = value.charset.toLowerCase();
454             if(value.charset.substr(0,4)=="win-"){
455                 value.charset = "windows-"+value.charset.substr(4); 
456             }else if(value.charset.match(/^utf\d/)){
457                 value.charset = "utf-"+value.charset.substr(3); 
458             }else if(value.charset.match(/^(us\-)?ascii$/)){
459                 value.charset = "utf-8"; 
460             }  
461             this.currentNode.meta.charset = value.charset;
462         }
463         if(value.format){
464             this.currentNode.meta.textFormat = value.format;
465         }
466         if(value.boundary){
467             this.currentNode.meta.mimeBoundary = value.boundary;
468         }
469         if(value.name && !this.currentNode.meta.fileName){
470             this.currentNode.meta.fileName = this.replaceMimeWords(value.name);
471         }
472         if(value.boundary){
473             this.currentNode.meta.mimeBoundary = value.boundary;
474             this.multipartTree.push({
475                 boundary: value.boundary,
476                 node: this.currentNode
477             });
478         }
479     } 
480     
481     return value;
482 };

484 /**
485  * Parses Content-Disposition header field value
486  * 
487  * Fetches filename to current node meta object
488  * 
489  * @param {String} value A Content-Disposition header field
490  */
491 MailParser.prototype.parseContentDisposition = function(value){
492     var returnValue = {};
493     
494     value = this.parseHeaderLineWithParams(value);
495     
496     if(value){
497         if(value.defaultValue){
498             this.currentNode.meta.contentDisposition = value.defaultValue.trim().toLowerCase();
499         }
500         if(value.filename){
501             this.currentNode.meta.fileName = this.replaceMimeWords(value.filename);
502         }
503     }

506 /**
507  * Parses the priority of the e-mail
508  * 
509  * @param {String} value The priority value
510  * @returns {String} priority string low|normal|high
511  */
512 MailParser.prototype.parsePriority = function(value){
513     value = value.toLowerCase().trim();
514     if(!isNaN(value)){
515         value = Number(value) || 0;
516         if(value == 3){
517             return "normal";
518         }else if(value > 3){
519             return "low";
520         }else{
521             return "high";
522         }
523     }else{
524         switch(value){
525             case "non-urgent":
526             case "low":
527                 return "low";
528             case "urgent":
529             case "hight":
530                 return "high";
531         }
532     }
533     return "normal";
534 };

536 /**
537  * Processes a line in text/html or text/plain node
538  * 
539  * Append the line to the content property
540  * 
541  * @param {String} line A line to be processed 
542  */
543 MailParser.prototype.handleTextLine = function(line){
544     
545     if(["quoted-printable", "base64"].indexOf(this.currentNode.meta.transferEncoding)>=0){
546         if(typeof this.currentNode.content != "string"){
547             this.currentNode.content = line;
548         }
549         this.currentNode.content += "\n"+line;
550     }else{
551         if(this.currentNode.meta.textFormat != "flowed"){
552             if(typeof this.currentNode.content != "string"){
553                 this.currentNode.content = this.encodeString(line);
554             }else{
555                 this.currentNode.content += "\n" + this.encodeString(line);
556             }
557         }else{
558             if(typeof this.currentNode.content != "string"){
559                 this.currentNode.content = this.encodeString(line);
560             }else if(this.currentNode.content.match(/[ ]{1,}$/)){
561                 this.currentNode.content += this.encodeString(line);
562             }else{
563                 this.currentNode.content += "\n"+this.encodeString(line);
564             }
565         }
566     }  
567 };

569 /**
570  * Processes a line in an attachment node
571  * 
572  * If a stream is set up for the attachment write the line to the
573  * stream as a Buffer object, otherwise append it to the content property
574  * 
575  * @param {String} line A line to be processed 
576  */
577 MailParser.prototype.handleAttachmentLine = function(line){
578     if(!this.currentNode.attachment){
579         return;
580     }
581     if(this.currentNode.stream){
582         this.currentNode.stream.write(new Buffer(line, "binary"));
583     }else if("content" in this.currentNode){
584         this.currentNode.content += line;
585     }
586 };

588 /**
589  * Finalizes a node processing
590  * 
591  * If the node is a text/plain or text/html, convert it to UTF-8 encoded string
592  * If it is an attachment, convert it to a Buffer or if an attachment stream is
593  * set up, close the stream
594  */
595 MailParser.prototype.finalizeContents = function(){
596     if(this.currentNode.content){
597         
598         if(!this.currentNode.attachment){
599             if(this.currentNode.meta.transferEncoding == "quoted-printable"){
600                 this.currentNode.content = mimelib.decodeQuotedPrintable(this.currentNode.content, false, this.currentNode.meta.charset);
601             }else if(this.currentNode.meta.transferEncoding == "base64"){
602                 this.currentNode.content = mimelib.decodeBase64(this.currentNode.content, false, this.currentNode.meta.charset);
603             }
604         }else{
605             if(this.currentNode.meta.transferEncoding == "quoted-printable"){
606                 this.currentNode.content = mimelib.decodeQuotedPrintable(this.currentNode.content, false, this.currentNode.meta.charset);
607                 this.currentNode.content = this.convertString(this.currentNode.content, "utf-8", this.currentNode.meta.charset);
608             }else if(this.currentNode.meta.transferEncoding == "base64"){
609                 this.currentNode.content = new Buffer(this.currentNode.content.replace(/[^\w\+\/=]/g,''), "base64");
610             }else{
611                 this.currentNode.content = new Buffer(this.currentNode.content, "binary");
612             }
613         }
614         
615     }
616 
617     if(this.currentNode.stream){
618         this.currentNode.stream.end();
619     }
620 };

622 /**
623  * Processes the mime tree
624  * 
625  * Finds text parts and attachments from the tree. If there's several text/plain
626  * or text/html parts, push the ones from the lower parts of the tree to the
627  * alternatives array
628  * 
629  * Emits "end" when finished
630  */
631 MailParser.prototype.processMimeTree = function(){
632     var level = 0, htmlLevel, textLevel, html, text,
633         returnValue = {}, i, len;
634     
635     this.mailData = {html:[], text:[], alternatives:[], attachments:[]};
636     
637     if(!this.mimeTree.meta.mimeMultipart){
638         this.processMimeNode(this.mimeTree, 0);
639     }else{
640         this.walkMimeTree(this.mimeTree);
641     }
642     
643     if(this.mailData.html.length){
644         for(i=0, len=this.mailData.html.length; i<len; i++){
645             if(!returnValue.html || this.mailData.html[i].level < htmlLevel){
646                 if(html){
647                     if(!returnValue.alternatives){
648                         returnValue.alternatives = [];
649                     }
650                     returnValue.alternatives.push({
651                         contentType: "text/html",
652                         content: html
653                     });
654                 }
655                 htmlLevel = this.mailData.html[i].level;
656                 html = this.mailData.html[i].content;
657             }
658         }
659     }
660     
661     if(this.mailData.text.length){
662         for(i=0, len=this.mailData.text.length; i<len; i++){
663             if(!returnValue.text || this.mailData.text[i].level < textLevel){
664                 if(text){
665                     if(!returnValue.alternatives){
666                         returnValue.alternatives = [];
667                     }
668                     returnValue.alternatives.push({
669                         contentType: "text/plain",
670                         content: text
671                     });
672                 }
673                 textLevel = this.mailData.text[i].level;
674                 text = this.mailData.text[i].content;
675             }
676         }
677     }
678     
679     returnValue.header = this.mimeTree.headers;
680     
681     if(this.mimeTree.subject){
682         returnValue.subject = this.mimeTree.subject;        
683     }
684     
685     if(this.mimeTree.from){
686         returnValue.from = this.mimeTree.from;        
687     }
688     
689     if(this.mimeTree.to){
690         returnValue.to = this.mimeTree.to;        
691     }
692     
693     if(this.mimeTree.cc){
694         returnValue.cc = this.mimeTree.cc;        
695     }
696     
697     if(this.mimeTree.bcc){
698         returnValue.bcc = this.mimeTree.bcc;        
699     }
700     
701     if(html){
702         returnValue.html = html;
703     }
704     
705     if(text){
706         returnValue.text = text;
707     }
708     
709     if(this.mailData.attachments.length){
710         returnValue.attachments = [];
711         for(i=0, len=this.mailData.attachments.length; i<len; i++){
712             returnValue.attachments.push(this.mailData.attachments[i].content);
713         }
714     }
715 
716     process.nextTick(this.emit.bind(this, "end", returnValue));
717 };

719 /**
720  * Walks the mime tree and runs processMimeNode on each node of the tree
721  * 
722  * @param {Object} node A mime tree node
723  * @param {Number} [level=0] current depth
724  */
725 MailParser.prototype.walkMimeTree = function(node, level){
726     level = level || 1;
727 
728     for(var i=0, len = node.childNodes.length; i<len; i++){
729         this.processMimeNode(node.childNodes[i], level);
730         this.walkMimeTree(node.childNodes[i], level+1);
731     }
732 };

734 /**
735  * Processes of a node in the mime tree
736  * 
737  * Pushes the node into appropriate this.mailData array (text/html to this.mailData.html array etc)
738  * 
739  * @param {Object} node A mime tree node
740  * @param {Number} [level=0] current depth
741  */
742 MailParser.prototype.processMimeNode = function(node, level){
743     level = level || 0;
744 
745     if(!node.attachment){
746         switch(node.meta.contentType){
747             case "text/html":
748                 this.mailData.html.push({content: node.content || "", level: level});
749                 break;
750             case "text/plain":
751                 this.mailData.text.push({content: node.content || "", level: level});
752                 break;
753         }
754     }else{
755         node.meta = node.meta || {};
756         if(node.content){
757             node.meta.content = node.content;
758         }
759         this.mailData.attachments.push({content: node.meta || {}, level: level});
760     }  
761 };

763 /**
764  * Converts a string from one charset to another
765  * 
766  * @param {Buffer|String} value A String to be converted
767  * @param {String} fromCharset source charset
768  * @param {String} [toCharset="UTF-8"] destination charset
769  * @returns {Buffer} Converted string as a Buffer (or SlowBuffer)
770  */
771 MailParser.prototype.convertString = function(value, fromCharset, toCharset){
772     toCharset = (toCharset || "utf-8").toUpperCase();
773     fromCharset = (fromCharset || "utf-8").toUpperCase();
774     
775     value = typeof value=="string"?new Buffer(value, "binary"):value;
776     
777     if(toCharset == fromCharset){
778         return value;
779     }
780     
781     try{ // in case there is no such charset or EINVAL occurs leave the string untouched
782         if(!this.iconv[fromCharset+toCharset]){
783             this.iconv[fromCharset+toCharset] = new Iconv(fromCharset, toCharset+'//TRANSLIT//IGNORE');
784         }
785         value = this.iconv[fromCharset+toCharset].convert(value);
786     }catch(E){}
787     
788     return value;
789 };

791 /**
792  * Encodes a header string to UTF-8
793  * 
794  * @param {String} value String to be encoded
795  * @returns {String} UTF-8 encoded string
796  */
797 MailParser.prototype.encodeString = function(value){
798     value = this.replaceMimeWords(this.convertString(value, this.currentNode.meta.charset).toString("utf-8"));
799     return value;
800 };

802 /**
803  * Replaces mime words in a string with UTF-8 encoded strings
804  * 
805  * @param {String} value String to be converted
806  * @returns {String} converted string
807  */
808 MailParser.prototype.replaceMimeWords = function(value){
809     return value.
810         replace(/(=\?[^?]+\?[QqBb]\?[^?]+\?=)\s+(?==\?[^?]+\?[QqBb]\?[^?]+\?=)/g, "$1"). // join mimeWords
811         replace(/\=\?[^?]+\?[QqBb]\?[^?]+\?=/g, (function(a){
812             return mimelib.decodeMimeWord(a);
813         }).bind(this));
814 };

816 /**
817  * Removes enclosing quotes ("", '', <>) from a string
818  * 
819  * @param {String} value String to be converted
820  * @returns {String} converted string
821  */
822 MailParser.prototype.trimQuotes = function(value){
823     value = (value || "").trim();
824     if((value.charAt(0)=='"' && value.charAt(value.length-1)=='"') || 
825       (value.charAt(0)=="'" && value.charAt(value.length-1)=="'") || 
826       (value.charAt(0)=="<" && value.charAt(value.length-1)==">")){
827         value = value.substr(1,value.length-2);
828     }
829     return value;
830 };

832 /**
833  * Generates a context unique filename for an attachment
834  * 
835  * If a filename already exists, append a number to it
836  * 
837  *     file.txt
838  *     file-1.txt
839  *     file-2.txt
840  * 
841  * @param {String} fileName source filename
842  * @returns {String} generated filename
843  */
844 MailParser.prototype.generateFileName = function(fileName){
845     var ext;
846     
847     // remove path if it is included in the filename
848     fileName = (fileName || "attachment").toString().split(/[\/\\]+/).pop().replace(/^\.+/,"") || "attachment";
849     
850     if(fileName in this.fileNames){
851         this.fileNames[fileName]++;
852         ext = fileName.substr((fileName.lastIndexOf(".") || 0)+1);
853         if(ext == fileName){
854             fileName += "-" +  this.fileNames[fileName];
855         }else{
856             fileName = fileName.substr(0, fileName.length - ext.length - 1) + "-" + this.fileNames[fileName] + "." + ext;
857         }
858     }else{
859         this.fileNames[fileName] = 0;
860     }
861     return fileName;  
862 };
